# Helper to solve math problems
def evaluate_math_expression(expr_str):
    """Safely evaluate a simple math expression string"""
    try:
        # Clean the expression - remove MathML tags
        import re
        # Extract content between math tags
        matches = re.findall(r'<math>(.*?)</math>', expr_str, re.DOTALL)
        if matches:
            expr = matches[0]
        else:
            expr = expr_str
            
        # Remove other tags and clean
        expr = re.sub(r'<.*?>', '', expr)
        expr = expr.replace(',', '').strip()
        
        # Try to evaluate
        return eval(expr, {"__builtins__": {}}, {"math": math})

        return None
