def check_mathml(text):
    """Check if MathML tags are properly formatted (not HTML-escaped)"""
    if not text:
        return True, "No MathML"
    
    # Check for escaped tags
    if '&lt;math' in text or '&gt;' in text or '&amp;' in text:
        return False, "Contains HTML-escaped characters"
    
    # Check for proper MathML structure
    if '<math>' in text and '</math>' in text:
        return True, "Proper MathML tags"
    elif '<math' in text:  # Might have attributes
        return True, "MathML with attributes"
