# Let's first check metadata consistency
def verify_metadata(data):
    issues = []
    
    # Check total questions
    total_questions = len(data["questions"])
    metadata_total = data["test_metadata"]["total_questions"]
    
    if total_questions != metadata_total:
        issues.append(("Metadata", "Total Questions Mismatch", 
                      f"Metadata claims {metadata_total} questions, but array has {total_questions}"))
    
    # Check difficulty distribution
    difficulty_counts = Counter([q["difficulty"].lower() for q in data["questions"]])
    metadata_diff = data["test_metadata"]["difficulty_distribution"]
    
    for diff in ["easy", "medium", "hard"]:
        actual = difficulty_counts.get(diff, 0)
        claimed = metadata_diff.get(diff, 0)
        if actual != claimed:
            issues.append(("Metadata", f"Difficulty Distribution - {diff}",
                          f"Claimed {claimed}, actual {actual}"))
    
    # Check question type distribution
    type_counts = Counter([q["question_type"] for q in data["questions"]])
    metadata_types = data["test_metadata"]["question_type_distribution"]
    
    # Map question_type to metadata keys
    type_mapping = {"MCQ": "multiple_choice", "SPR": "student_produced_response"}
    
    for q_type, count in type_counts.items():
        metadata_key = type_mapping.get(q_type)
        if metadata_key:
            claimed = metadata_types.get(metadata_key, 0)
            if count != claimed:
                issues.append(("Metadata", f"Question Type Distribution - {q_type}",
                              f"Claimed {claimed}, actual {count}"))
    
    # Check focus domains and skills
    all_domains = set([q["domain"] for q in data["questions"]])
    focus_domains = set(data["test_metadata"]["focus_domains"])
    
    missing_in_focus = all_domains - focus_domains
    if missing_in_focus:
        issues.append(("Metadata", "Focus Domains Incomplete",
                      f"Questions include domains not in focus: {missing_in_focus}"))
    
    all_skills = set([q["skill"] for q in data["questions"]])
    focus_skills = set(data["test_metadata"]["focus_skills"])
    
    missing_skills = all_skills - focus_skills
    if missing_skills:
        issues.append(("Metadata", "Focus Skills Incomplete",
                      f"Questions include skills not in focus: {missing_skills}"))
    
    return issues
